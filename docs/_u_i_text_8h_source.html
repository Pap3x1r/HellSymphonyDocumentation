73 std::cerr << "UIText render setup error" << std::endl;
74 return;

77 // Enable transparency
78 glEnable(GL_BLEND);
79 glBlendFunc(GL_SRC_ALPHA, GL_ONE_MINUS_SRC_ALPHA);
80
81 // Set uniforms
82 glm::mat4 currentMatrix = transform.getTransformMat4();
83 currentMatrix = glm::translate(currentMatrix, textOffset);
84 currentMatrix = globalModelTransform * currentMatrix;
85
86 glUniformMatrix4fv(modelMatrixId, 1, GL_FALSE, glm::value_ptr(currentMatrix));
87 glUniform1i(renderModeId, 1); // Textured render mode
88 glUniform4f(colorId, 1.0f, 1.0f, 1.0f, 1.0f); // Pure white color
89 glUniform1f(alphaOverrideId, alpha);

130 void loadText(string text, SDL_Color textColor, int fontSize, bool bold = false) {
131 this->text = text;
132 this->color = textColor;
133 this->fontSize = fontSize;
134 //this->alpha = static_cast<float>(textColor.a) / 255.0f;
135
136 TTF_Font* font = nullptr;
137
138 if (bold) {
139 font = TTF_OpenFont("../Resource/Font/EBGaramond-Bold.ttf", fontSize);
140 }
141 else {
142 font = TTF_OpenFont("../Resource/Font/EBGaramond.ttf", fontSize);
143 }
144
145 if (!font) {
146 cerr << "Failed to load font: " << TTF_GetError() << endl;
147 return;
148 }
149
150 SDL_Surface* textSurface = TTF_RenderText_Blended(font, text.c_str(), textColor);
151 if (!textSurface) {
152 cerr << "Failed to render text: " << TTF_GetError() << endl;
153 TTF_CloseFont(font);
154 return;
155 }
156
157 int potW = 1;
158 int potH = 1;
159 while (potW < textSurface->w) potW <<= 1;
160 while (potH < textSurface->h) potH <<= 1;
161
162 SDL_Surface* potSurface = SDL_CreateRGBSurface(0, potW, potH, 32,
163 0x00FF0000, 0x0000FF00, 0x000000FF, 0xFF000000);
164
165 if (!potSurface) {
166 cerr << "Failed to create POT surface." << endl;
167 SDL_FreeSurface(textSurface);
168 TTF_CloseFont(font);
169 return;
170 }
171
172 SDL_BlitSurface(textSurface, nullptr, potSurface, nullptr);
173
174 if (texture != 0) {
175 glDeleteTextures(1, &texture);
176 }
177
178 glGenTextures(1, &texture);
179 glBindTexture(GL_TEXTURE_2D, texture);
180
181 glPixelStorei(GL_UNPACK_ALIGNMENT, 1);
182 glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_MIN_FILTER, GL_LINEAR);
183 glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_MAG_FILTER, GL_LINEAR);
184 glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_WRAP_S, GL_CLAMP_TO_EDGE);
185 glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_WRAP_T, GL_CLAMP_TO_EDGE);
186
187 glTexImage2D(GL_TEXTURE_2D, 0, GL_RGBA, potW, potH, 0, GL_RGBA, GL_UNSIGNED_BYTE, potSurface->pixels);
188
189 glm::vec3 currentScale = getTransform().getScale();
190 if (currentScale == glm::vec3(1.0f)) {
191 float w = textSurface->w / 100.0f;
192 float h = textSurface->h / 100.0f;
193 this->getTransform().setScale(glm::vec3(w, h, 1.0f));
194 }
195
196 SDL_FreeSurface(textSurface);
197 SDL_FreeSurface(potSurface);
198 TTF_CloseFont(font);
199 }
